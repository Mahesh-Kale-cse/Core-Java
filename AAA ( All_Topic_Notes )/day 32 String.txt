length -> counts the no of characters in the string
String s = "Yash";
System.out.println(s.length()); //4

charAt -> gives the character at given index
String s = "Yash";
System.out.println(s.charAt(1)); //a

codePointAt -> gives ASCII value of the character at given index
String s = "Yash";
System.out.println(s.codePointAt(1)); //97

contains -> checks if the string contains given set of characters
String s = "Vaishnavi";
System.out.println(s.contains("bh")); //false
System.out.println(s.contains("sh")); //true

equals -> checks for the equality of 2 strings - case sensitive
String s = "Java";
System.out.println(s.equals("C++")); //false
System.out.println(s.equals("Java")); //true

equalsIgnoreCase -> checks for the equality of 2 strings - case insensitive
String s = "Java";
System.out.println(s.equalsIgnoreCase("C++")); //false
System.out.println(s.equalsIgnoreCase("Java")); //true
System.out.println(s.equalsIgnoreCase("JAVA")); //true

startsWith -> checks if the string is starting with given characters or not
String s = "Java";
System.out.println(s.startsWith("Ja")); //true
System.out.println(s.startsWith("Ab")); //false

endsWith -> checks if the string is ending with given characters or not
String s = "Java";
System.out.println(s.startsWith("va")); //true
System.out.println(s.startsWith("sm")); //false

indexOf -> gives the index of the given character in the string
String s = "Java";
System.out.println(s.indexOf('v')); //2

lastIndexOf -> gives the last index of the given character in the string
String s = "Java";
System.out.println(s.lastIndexOf('a')); //3
 
replace -> replaces the old with new
String s = "Java";
System.out.println(s.replace('a', '@')); //J@v@
System.out.println(s.replace('y', '$'));//Java

substring -> reads the part of the string basis on the indices.
String s = "international";
System.out.println(s.substring(5, 13));//national
System.out.println(s.substring(10));//nal

repeat -> repeats the string given number of times
String s = "Omkar";
System.out.println(s.repeat(3));//OmkarOmkarOmkar

concat -> joins the given string to the string
String s = "Success";
System.out.println(s.concat("fully"));//Successfully

split -> makes the pieces of the string basis on delimiter
String s = "I am Java developer";
String[] m = s.split("a");

for (int i = 0; i < m.length; i++) {
	System.out.println(m[i]);
	}
//I 
//m J
//v
// developer

trim -> removes leading and trailing white spaces. intermediate white spaces are not removed.
String s = "        Java        Developer         ";
System.out.println(s.trim());//Java        Developer
